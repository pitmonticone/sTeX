  \begin{smodule}[title=Notation Components]{Comp}
    \usemodule{concepts?Symbol}
    \usemodule{concepts?Notation}
    \usemodule{concepts?SemanticMacro}
    \importmodule{macros?comp}

    \begin{sparagraph}[style={defibox,symdoc},name=comp]
        The \cs{comp}-macro marks components to be highlighted in a \sn{notation} 
        for a \sn{symbol} taking (one or more) arguments.
    
        This is necessary because
        it is (nearly) impossible for \LaTeX\ to figure out,
        which parts of a \sn{notation} to highlight and which not
         on its own --
        in particular, the highlighting should stop for the 
        \emph{arguments} of a \sn{semantic macro}.
    
        \Sns{notation} that do not take arguments, as well as
        \sns{operator notation}, are automatically wrapped in
        \cs{comp}.
    \end{sparagraph}
    
  \end{smodule}
