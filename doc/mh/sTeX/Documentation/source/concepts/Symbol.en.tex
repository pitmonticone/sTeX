\begin{document}
  \begin{smodule}[title=\sTeX-Symbols]{Symbol}
    \importmodule{macros?symdecl}
    \usemodule{concepts?Module}

    \begin{sparagraph}[style={defibox,symdoc},name=symbol]
        A \definame{symbol} is a \emph{named} 
        concept that can be defined, documented and referenced.

        Examples for \sns{symbol} are mathematical constants, 
        functions, theorems, statements, principles -- anything
        that has a (somewhat) precise meaning and can be referenced
        by name can be a \sn{symbol}.
        
        \Sns{symbol} can be declared using the \cs{symdecl}-command.
    \end{sparagraph}
    
  \end{smodule}
\end{document}
