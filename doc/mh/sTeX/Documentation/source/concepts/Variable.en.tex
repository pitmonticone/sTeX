    \usemodule{macros?notation}

    \begin{sparagraph}[style={defibox,symdoc},name=variable]
        A \definame{variable} represents a \emph{generic}
        or \emph{unspecified} object.

        \Sns{variable} can be declared using the \cs{vardef}-macro,
        whose syntax is analogous to \cs{symdef}.
        
        Note that \sns{variable} are local to the current \TeX-group
        (e.g. environment).
    \end{sparagraph}
    
  \end{smodule}
