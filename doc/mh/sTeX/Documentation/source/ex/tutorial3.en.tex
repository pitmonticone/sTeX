\begin{smodule}{Monoid}
  \symdecl*{monoid}
  \symdef{universe}{\comp{M}}
  \symdef{op}[args=2,op=\comp{\circ}]{#1 \mathbin{\comp{\circ}} #2}
  \symdef{unit}{\comp{e}}
  
  \begin{definition}
    A \textbf{\symname{monoid}} is a \symref{mathstruct}{structure} 
    $\mathstruct{\universe,\op!,\unit}$ with $\op!: M\times M\to M$, such
    that $\op!$ is associative and $\op{x}{\unit} = \op{\unit}{x} = x$ 
    for all $x \in \universe$.
  \end{definition}
\end{smodule}
