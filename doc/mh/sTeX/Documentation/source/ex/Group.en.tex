    \begin{sdefinition}[for={group,inverse}]

      \sr{mathstruct}{structure} $\group!$

      $\inlineass[name=inverse_axiom]{
        \eq{
          \op{\vx,\inverse{\vx}},
          \op{\inverse{\vx},\vx},
          \unit
        }
      }$ for all $\inset{\vx}{\universe}$.
    \end{sdefinition}
  \end{extstructure}
