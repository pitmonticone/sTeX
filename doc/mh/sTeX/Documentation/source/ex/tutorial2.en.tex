\begin{smodule}{Monoid}
  \symdecl*{monoid}
  \begin{definition}
    A \textbf{\symname{monoid}} is a structure $\langle M,\circ,e\rangle$
    with $\circ:M\times M\to M$, such that $\circ$ is associative and
    $x\circ e=e \circ x = x$ for all $x\in M$.
  \end{definition}
\end{smodule}
