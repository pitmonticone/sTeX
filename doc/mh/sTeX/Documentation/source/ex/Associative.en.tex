\begin{smodule}{Associative}
  \importmodule[sTeX/MathBase/Functions]{mod?Function}
  \importmodule[sTeX/MathBase/Relations]{mod?Equal}

  \symdecl{associative}[args=1]

  \begin{sdefinition}[for=associative]
    \vardef{vU}[name=U]{U}
    \vardef{op}[args=2,type=\funspace{\vU,\vU}\vU,bind=forall,
      op=\circ]
      {#1 \mathbin{\comp{\circ}} #2}
    \vardef{vx}[type=\vU,bind=forall]{x}
    \vardef{vy}[type=\vU,bind=forall]{y}
    \vardef{vz}[type=\vU,bind=forall]{z}

    A \symname{function} $\fun{\op!}{\vU,\vU}\vU$ is 
    called \definame{associative}, if
    \definiens{$
      \eq{
          \op{ (\op{\vx}{\vy}) }{\vz},
          \op{\vx}{ (\op{\vy}{\vz}) }
        }
    $}
    for all $\inset{\vx,\vy,\vz}\vU$.
  \end{sdefinition}
\end{smodule}
