\begin{smodule}{GroupDivision}
  \importmodule{ex?Group}

  \begin{extstructure*}{group}
    \begin{sdefinition}[macro=div,name=division,args=2]
      \notation{div}[op=/]{#1 \mathbin{\comp{/}} #2}
      \vardef{vx}[name=x,bind=forall,type=\universe]{x}
      \vardef{vy}[name=y,bind=forall,type=\universe]{y}

      For a \sn{group} $\group![universe,op,unit]$,
      we define \definame{div} $\defnotation{\div{\vx}{\vy}}$ as
      $\definiens{\op{\vx,\inverse{\vy}}}$.
    \end{sdefinition}
  \end{extstructure*}
\end{smodule}
