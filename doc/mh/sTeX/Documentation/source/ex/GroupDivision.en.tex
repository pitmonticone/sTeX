\begin{smodule}{GroupDivision}
  \importmodule{ex?Group}

  \begin{extstructure*}{group}
    \begin{sdefinition}[macro=div,name=division,args=2]
      \notation{div}[op=\comp{/}_{\this}]
        {#1 \mathbin{\comp{/}_{\this}} #2}
      \vardef{vx}[name=x,bind=forall]{\comp{x}}
      \vardef{vy}[name=y,bind=forall]{\comp{y}}

      For a \sn{group} $\group![universe,op,unit]$,
      we define \definame{div} $\div{\vx}{\vy}$ as
      $\definiens{\op{\vx,\inverse{\vy}}}$.
    \end{sdefinition}
  \end{extstructure*}
\end{smodule}
