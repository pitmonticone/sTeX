\begin{smodule}{Monoid}
  \importmodule[sTeX/MathBase/Functions]{mod?Function}
  \importmodule[sTeX/MathBase/Relations]{mod?Equal}

  \begin{mathstructure}{monoid}
    \symdef{universe}[type=\collection]{M}
    \symdef{op}[name=operation,
      type=\funspace{\universe,\universe}\universe,
      args=a,op=\circ]{#1}
      {##1 \mathbin{\comp{\circ}} ##2}
    \symdef{unit}[type=\universe]{e}
  
    \begin{sdefinition}[for={monoid,op,unit}]
      \vardef{vx}[name=x,type=\universe]{x}
      A \definame{monoid} is a \symref{mathstruct}{structure} 
      $\mathstruct{\universe,\op!,\unit}$ with 
      $\fun{\op!}{\universe,\universe}{\universe}$, such that $\op!$ 
      is associative and $\eq{\op{\vx,\unit},\op{\unit,\vx},\vx}$ 
      for all $\inset{\vx}{\universe}$.
    \end{sdefinition}
  \end{mathstructure}
\end{smodule}
