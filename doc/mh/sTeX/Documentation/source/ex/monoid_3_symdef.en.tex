\begin{smodule}{Monoid}
  \symdecl*{monoid}
  \symdef{universe}{M}
  \symdef{op}[name=operation,args=2,op=\circ]
    {#1 \mathbin{\comp{\circ}} #2}
  \symdef{unit}{e}
  
  \begin{definition}
    A \textbf{\symname{monoid}} is a \symref{mathstruct}{structure} 
    $\mathstruct{\universe,\op!,\unit}$ with 
    $\op!: \universe \times \universe \to \universe$, such
    that $\op!$ is associative and 
    $\op{x}{\unit}=\op{\unit}{x}=x$ 
    for all $x\in \universe$.
  \end{definition}
\end{smodule}
