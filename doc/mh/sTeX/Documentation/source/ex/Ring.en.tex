\begin{smodule}{Ring}
  \importmodule{ex?AbelianGroup}
  \symdecl*{distributive law}
  \begin{mathstructure}{ring}
    \begin{copymodule}{addition}{abgroup}
      \renamedecl{universe}{universe}
      \renamedecl{op}[addition]{plus}
      \renamedecl{unit}[zero]{zero}
      \renamedecl{inverse}[negation]{uminus}
    \end{copymodule}
    \notation*{universe}[op=\comp{R}_{\this}]{\comp{R}_{\this}}
    \notation*{plus}[op=\comp{+}_{\this}]
      {#1}{##1 \mathbin{\comp{+}_{\this}} ##2}
    \notation*{zero}[op=\comp{0}_{\this}]{\comp{0}_{\this}}
    \notation*{uminus}[op=\comp{-}_{\this}]{\comp{-}_{\this}}

    \begin{copymodule}{multiplication}{monoid}
      \assign{universe}{\symname{ring?universe}}
      \renamedecl{op}[multiplication]{mult}
      \renamedecl{unit}[one]{one}
    \end{copymodule}
    \notation*{one}[op=\comp{1}_{\this}]{\comp{1}_{\this}}
    \notation*{mult}[op=\comp{\cdot}_{\this}]
    {#1}{##1 \mathbin{\comp{\cdot}_{\this}} ##2}

    \begin{sdefinition}
      \vardef{vx}[name=x,bind=forall,type=\universe]{\comp{x}}
      \vardef{vy}[name=y,bind=forall,type=\universe]{\comp{y}}
      \vardef{vz}[name=z,bind=forall,type=\universe]{\comp{z}}
      A \definame{ring} is a \sr{mathstruct}{structure} 
      $\ring!$ such that $\ring![universe,plus,zero,uminus]$
      is an \symname{abgroup}, $\ring![universe,mult,one]$
      is a \symname{monoid}, and the following 
      \definame[post=s]{distributive law}
      hold for all $\inset{\vx,\vy,\vz}{\universe}$:
      \begin{enumerate}
        \item \inlineass[name=left distributive]{$\eq{
          \mult{\vx,\plus{\vy,\vz}},
          \plus{\mult{\vx,\vy},\mult{\vx,\vz}}
        }$} and
        \item \inlineass[name=right distributive]{$\eq{
          \mult{\plus{\vx,\vy},\vz},
          \plus{\mult{\vx,\vz},\mult{\vy,\vz}}
        }$} 
      \end{enumerate}
    \end{sdefinition}
  \end{mathstructure}
\end{smodule}
