\begin{smodule}{Ring}
  \importmodule{ex?AbelianGroup}
  \symdecl*{distributive law}
  \begin{mathstructure}{ring}

    \copymod{additive group}{abgroup}{
      universe@universe,
      op@[addition]plus,
      unit@[zero]zero,
      inverse@[negation]uminus
    }
    \notation*{universe}{R}
    \notation*{plus}[op=+,prec=100]
      {#1}{##1 \mathbin{\comp{+}} ##2}
    \notation*{zero}{0}
    \notation*{uminus}[op=-]{\comp{-}#1}

    \copymod{multiplicative monoid}{monoid}{
      universe=\symname{ring-module?universe},
      op@[multiplication]mult,
      unit@[one]one
    }
    \notation*{one}{1}4
    \notation*{mult}[op=\cdot,prec=50]
    {#1}{##1 \mathbin{\comp{\cdot}} ##2}

    \begin{sdefinition}
      \vardef{vx}[name=x,bind=forall,type=\universe]{x}
      \vardef{vy}[name=y,bind=forall,type=\universe]{y}
      \vardef{vz}[name=z,bind=forall,type=\universe]{z}
      A \definame{ring} is a \sr{mathstruct}{structure} 
      $\ring!$ such that $\ring![universe,plus,zero,uminus]$
      is an \symname{abgroup}, $\ring![universe,mult,one]$
      is a \symname{monoid}, and the following 
      \definame[post=s]{distributive law}
      hold for all $\inset{\vx,\vy,\vz}{\universe}$:
      \begin{enumerate}
        \item \inlineass[name=left distributive]{$\eq{
          \mult{\vx,\plus{\vy,\vz}},
          \plus{\mult{\vx,\vy},\mult{\vx,\vz}}
        }$} and
        \item \inlineass[name=right distributive]{$\eq{
          \mult{\plus{\vx,\vy},\vz},
          \plus{\mult{\vx,\vz},\mult{\vy,\vz}}
        }$} 
      \end{enumerate}
    \end{sdefinition}
  \end{mathstructure}
\end{smodule}
