  \begingroup 
  \catcode`\\=12\relax
  \catcode`\#=12\relax
  \catcode`\&=12\relax
  \catcode`\$=12\relax
  \catcode`\^=12\relax
  \catcode`\_=12\relax
  \catcode`\ =12\relax
  \catcode`^^J=12\relax
  \endlinechar=`^^J
  \newlinechar=-1
%^^A    \everyeof{\noexpand}
  \example_a:n
}
\long\def\example_a:n #1 {
  \endgroup
  \begin{example@border}
    \immediate\openout\testoutfile=stextest.tst
    \immediate\write\testoutfile{
      \c_backslash_str begin{latexcode}
      \detokenize{^^J}#1
      \c_backslash_str end{latexcode}
    }
    \immediate\closeout\testoutfile
    \begin{stextest@input}
      \catcode`\#=12\relax
      \input{stextest.tst}
    \end{stextest@input}
    \immediate\openout\testoutfile=stextest.tst
    \immediate\write\testoutfile{#1}
    \immediate\closeout\testoutfile
    \begin{stextest@output}
      \input{stextest.tst}
    \end{stextest@output}
  \end{example@border}
}
